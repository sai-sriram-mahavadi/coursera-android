39  FI_FABRIC_SERVER + switch + "/slot-"+

41  ucs.FabricDceSwSrvEp.PORT_ID: server_port})
42 
43  # Configure Uplink Port
44  # will take params port_id, slot_id
45  def configure_uplink_port(self, uplink_port, switch, slot_id):
46  handle = self.handle
47  obj = handle.GetManagedObject(None, None, {"dn": "fabric/lan/A"})
48  FIUtils.addOrIgnoreMO(obj, ucs.FabricEthLanEp.ClassId(),
49  {ucs.FabricEthLanEp.DN:
50  FI_FABRIC_LAN + "phys-slot-" + slot_id +
51  "-port-" + uplink_port,

53  ucs.FabricEthLanEp.SLOT_ID: slot_id,
54  ucs.FabricEthLanEp.ADMIN_STATE: "enabled",
55  ucs.FabricEthLanEp.PORT_ID: uplink_port})
