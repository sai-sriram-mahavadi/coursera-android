131  self,

150  if 'subnet' in line and 'netmask' in line:
151  lines_to_write.append(
152  "subnet " +
153  subnet +
154  " netmask " +

156  " {\n")
157  elif 'range' in line:
158  lines_to_write.append(
159  "\trange " +

167  netmask +
168  ";\n")

170  lines_to_write.append(
171  "\t\toption broadcast-address\t\t" +
172  broadcast_ip +
173  ";\n")
174  elif 'option routers' in line:
175  lines_to_write.append(
176  "\t\toption routers\t\t" +
177  gateway +
178  ";\n")
179  elif 'option domain-name' in line and 'option domain-name-servers' not in line:
180  lines_to_write.append(
181  "\t\toption domain-name\t\t\"" +
182  domain_name +
183  "\";\n")
184  elif 'option domain-name-servers' in line:
185  lines_to_write.append(
186  "\t\toption domain-name-servers\t\t" +
187  nameserver +
188  ";\n")
189  elif 'default-lease-time'in line:
190  lines_to_write.append(
191  "\tdefault-lease-time\t" +
192  lease_time +
193  ";\n")
194  elif 'max-lease-time' in line:
195  lines_to_write.append(
196  "\tmax-lease-time\t" +
197  max_lease_time +
198  ";\n")
199  else:
200  lines_to_write.append(line)
201  with open("/etc/dhcp/dhcpd.conf", "w") as file:
202  file.writelines(lines_to_write)
203  shell_command("systemctl start dhcpd.service")
